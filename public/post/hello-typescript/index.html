Class && Module

class Animal {
    constructor(public age: number) {
    }
    addAge(amount: number) {
        this.age += amount;
        return this.age;
    }
}

var Animal = (function () {
    function Animal() {
        this.age = 0;
    }
    Animal.prototype.addAge = function(amount) {
        this.age += amount;
        return this.age;
    };
    return Animal;
})();

module Hoge {
    var s = "hello";
    export function f() {
        return s;
    }
}

Hoge.f();
Hoge.s; // Error, s is not exported // variable 's' is private

var Hoge;
(function(exports) {
    var s = "hello";
    function f() {
        return s;
    }
    exports.f = f;
})(Hoge||(Hoge={}));
